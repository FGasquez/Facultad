#### lectura que puede atender a lo sumo 5 consultas simult√°neas.

      monitor DB
        condition queue;
        int sleep = 0;
        int count = 0;

        procedure in(x){
          if(count < 6){
            count ++;
          }else{
            sleep ++;
            wait(queue);
          }
        }

        procedure out(x){
          if(count < 6){
            count --;
          }else{
            sleep --;
            signal(queue);
          }
        }
